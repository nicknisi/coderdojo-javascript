Let's Get Started!

Start with a an html file

<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8">
    <title>Test Page</title>
  </head>
<body>
</body>
</html>

Add a script tag

<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8">
    <title>Test Page</title>
    <script>
      alert('Hello, World!');
    </script>
  </head>
<body>
</body>
</html>

Hello, World!

// create a variable to hold the text
var text = 'Hello, World!';

// output it to the console
console.log(text);
// aler the text
alert(text);
